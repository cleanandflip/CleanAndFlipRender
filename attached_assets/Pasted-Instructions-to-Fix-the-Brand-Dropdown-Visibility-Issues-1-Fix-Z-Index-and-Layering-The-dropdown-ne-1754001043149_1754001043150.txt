4. Complete Updated Component Styles
Here's the full styling solution:
tsx// In SearchableSelect component
const dropdownStyles = {
  container: "absolute z-50 w-full mt-1 max-h-60 overflow-auto",
  dropdown: "bg-gray-800 border-2 border-gray-500 rounded-md shadow-2xl",
  option: "w-full px-4 py-3 text-left text-gray-100 hover:bg-blue-600 hover:text-white transition-all duration-150 border-b border-gray-700 last:border-b-0",
  noResults: "px-4 py-3 text-gray-400 text-center",
  customOption: "w-full px-4 py-3 text-left text-blue-400 hover:bg-blue-600 hover:text-white transition-all duration-150"
};

// Apply styles
{isOpen && (
  <div className={dropdownStyles.container}>
    <div className={dropdownStyles.dropdown}>
      {filteredOptions.length > 0 ? (
        filteredOptions.map((option) => (
          <button
            key={option}
            type="button"
            onClick={() => handleSelect(option)}
            className={dropdownStyles.option}
          >
            {option}
          </button>
        ))
      ) : (
        <>
          {allowCustom ? (
            <button
              type="button"
              onClick={() => handleSelect(search)}
              className={dropdownStyles.customOption}
            >
              Use "{search}" (custom brand)
            </button>
          ) : (
            <div className={dropdownStyles.noResults}>
              No brands found
            </div>
          )}
        </>
      )}
    </div>
  </div>
)}
